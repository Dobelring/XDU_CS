IO244 EQU 0230H
IO273 EQU 0230H
STACK1 SEGMENT STACK
    DW 100 DUP(?)
STACK1 ENDS
 
DATA SEGMENT WORD PUBLIC 'DATA'
DATA ENDS
 
CODE SEGMENT
ASSUME CS:CODE, DS:DATA, SS:STACK1
START PROC NEAR
    MOV AX, DATA
    MOV DS, AX
 
INPUT:
    MOV DX, IO244
    IN AL, DX
    CMP AL, 0FFH ; 左到右
    JZ X1
    CMP AL, 000H ; 右到左
    JZ X2
    MOV DX, IO273
    OUT DX, AL
    JMP INPUT
 
X1:
    MOV AL, 0FEH
    MOV DX, IO273
LR:
    CALL DELAY
    OUT DX, AL
    ROL AL, 1
    CMP AL, 0FEH
    JNE LR
    JMP INPUT
 
X2:
    MOV AL, 07FH
    MOV DX, IO273
RL:
    CALL DELAY
    OUT DX, AL
    ROR AL, 1
    CMP AL, 07FH
    JNE RL
    JMP INPUT
 
DELAY PROC NEAR
Delay1:
    XOR CX, CX
    LOOP $
    RET
Delay ENDP
 
START ENDP
CODE ENDS
END START
